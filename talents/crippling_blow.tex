\newcommand\cripplingBlowTitle{ Crippling Blow}
\newcommand\cripplingBlowActive{true}
\newcommand\cripplingBlowRanked{false}
\newcommand\cripplingBlowDescription{ Increase the difficulty of next combat check by 1. If check deals damage, target suffers 1 Strain whenever he moves for the remainder of the encounter. }
\newcommand\cripplingBlowTalent{ \Talent[name=\cripplingBlowTitle, label=cripplingblow, active=\cripplingBlowActive, ranked=\cripplingBlowRanked]{
The character may voluntarily increase the difficulty
of a combat check by one to deal a crippling blow. If
he succeeds and deals damage to the target's wound
threshold, the target suffers one strain whenever he
moves for the remainder of the encounter.
} }
