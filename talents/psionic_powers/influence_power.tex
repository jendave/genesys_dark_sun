\newcommand\psionicInfluenceTitle{ Influence (Power)}
\newcommand\psionicInfluenceDescription{
    The character may attempt to guide, shape, and even twist the thoughts and feelings of others.
    The character may also spend \darklight to stress the mind of one living target he is engaged with, inflicting 1 strain.\\
    \textbf{Special Rule:} The psionic may only use \light to influence positive emotions and only \dark for negative emotions. Other emotions, such as confusion, may use \darklight.
}
\newcommand\psionicInfluenceTalent{ \Talent[name=\psionicInfluenceTitle, label=psionicinfluence, active=false, ranked=true]{\psionicInfluenceDescription} }

\newcommand\influenceDurationTitle{ Influence Duration}
\newcommand\influenceDurationDescription{ Spend \darklight to increase duration by number of rounds (or minutes) equal to Duration upgrades purchased. }
\newcommand\influenceDurationTalent{ \Talent[name=\influenceDurationTitle, label=psionicinfluenceduration, active=false, ranked=true]{\influenceDurationDescription} }

\newcommand\influenceControlATitle{ Influence Control 1}
\newcommand\influenceControlADescription{
    The Psionic may make an opposed Discipline vs Discipline
    checks combined with an Influence Power check . If the user
    spends \darklight and succeeds on the check, he can force the target
    to adopt an emotional state or believe something untrue,
    lasting for 1 round or 5 minutes.
}
\newcommand\influenceControlATalent{ \Talent[name=\influenceControlATitle, label=psionicinfluencecontrola, active=false, ranked=true]{\influenceControlADescription} }

\newcommand\influenceControlBTitle{ Influence Control 2}
\newcommand\influenceControlBDescription{
    When making a Coercion, Charm, Deception, Leadership, or Negotiation check, the Psionic
    may roll an Influence Power check as part of his dice pool. He may spend \darklight to gain \success or \advantage
    (user's choice) on the check.
}
\newcommand\influenceControlBTalent{ \Talent[name=\influenceControlBTitle, label=psionicinfluencecontrolb, active=false, ranked=true]{\influenceControlBDescription} }
